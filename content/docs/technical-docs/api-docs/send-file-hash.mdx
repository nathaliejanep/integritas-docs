```typescript
// Using fetch API
const API_KEY = 'your-api-key';
const REQUEST_ID = 'test-123';

async function sendFile(filePath: string) {
  const formData = new FormData();

  // For browser: use file input
  // const fileInput = document.querySelector('input[type="file"]');
  // formData.append('file', fileInput.files[0]);

  // For Node.js: use fs
  const fs = await import('fs');
  const file = fs.readFileSync(filePath);
  const blob = new Blob([file]);
  formData.append('file', blob, 'yourfile.docx');

  const response = await fetch(
    'https://integritas.minima.global/core/v1/file/hash',
    {
      method: 'POST',
      headers: {
        'x-api-key': API_KEY,
        'x-request-id': REQUEST_ID,
      },
      body: formData,
    }
  );

  const data = await response.json();
  console.log('Response:', data);
  return data;
}

// Usage
sendFile('./path/to/yourfile.docx');
```

  <Tab value='Python'>

```python
import requests

API_KEY = 'your-api-key'
REQUEST_ID = 'test-123'

def send_file(file_path):
    url = 'https://integritas.minima.global/core/v1/file/hash'

    headers = {
        'x-api-key': API_KEY,
        'x-request-id': REQUEST_ID,
    }

    with open(file_path, 'rb') as file:
        files = {'file': file}
        response = requests.post(url, headers=headers, files=files)

    data = response.json()
    print('Response:', data)
    return data

# Usage
send_file('./path/to/yourfile.docx')
```
